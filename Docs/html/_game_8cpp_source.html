207void Game::Update(const GameTimer& gt)
208{
209 OnKeyboardInput(gt);
210 //UpdateCamera(gt);
211
212 // Cycle through the circular frame resource array.
213 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
214 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
215
216 // Has the GPU finished processing the commands of the current frame resource?
217 // If not, wait until the GPU has completed commands up to this roof point.
218 if (mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
219 {
220 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
221 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
222 WaitForSingleObject(eventHandle, INFINITE);
223 CloseHandle(eventHandle);
224 }
225
226 UpdateObjectCBs(gt);
227 UpdateMaterialCBs(gt);
228 UpdateMainPassCB(gt);
229
230 world.update(gt, mAllRitems);

233void Game::Draw(const GameTimer& gt)
234{
235 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
236
237 // Reuse the memory associated with command recording.
238 // We can only reset when the associated command lists have finished execution on the GPU.
239 ThrowIfFailed(cmdListAlloc->Reset());
240
241 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
242 // Reusing the command list reuses memory.
243 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mPSOs["opaque"].Get()));
244
245 mCommandList->RSSetViewports(1, &mScreenViewport);
246 mCommandList->RSSetScissorRects(1, &mScissorRect);
247
248 // Indicate a state transition on the resource usage.
249 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
250 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
251
252 // Clear the back buffer and depth buffer.
253 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), (float*)&mMainPassCB.FogColor, 0, nullptr);
254 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
255
256 // Specify the buffers we are going to render to.
257 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
258
259 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
260 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
261
262 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
263
264 auto passCB = mCurrFrameResource->PassCB->Resource();
265 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
266
267 DrawRenderItems(mCommandList.Get(), mRitemLayer[(int)RenderLayer::Opaque]);
268
269 mCommandList->SetPipelineState(mPSOs["alphaTested"].Get());
270 DrawRenderItems(mCommandList.Get(), mRitemLayer[(int)RenderLayer::AlphaTested]);
271
272 mCommandList->SetPipelineState(mPSOs["transparent"].Get());
273 DrawRenderItems(mCommandList.Get(), mRitemLayer[(int)RenderLayer::Transparent]);
274
275 // Indicate a state transition on the resource usage.
276 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
277 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
278
279 // Done recording commands.
280 ThrowIfFailed(mCommandList->Close());
281
282 // Add the command list to the queue for execution.
283 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
284 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
285
286 // Swap the back and front buffers
287 ThrowIfFailed(mSwapChain->Present(0, 0));
288 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
289
290 // Advance the roof value to mark commands up to this roof point.
291 mCurrFrameResource->Fence = ++mCurrentFence;
292
293 // Add an instruction to the command queue to set a new roof point.
294 // Because we are on the GPU timeline, the new roof point won't be
295 // set until the GPU finishes processing all the commands prior to this Signal().
296 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
297}

338void Game::UpdateObjectCBs(const GameTimer& gt)
339{
340 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
341 for (auto& e : mAllRitems)
342 {
343 // Only update the cbuffer data if the constants have changed.
344 // This needs to be tracked per frame resource.
345 if (e->NumFramesDirty > 0)
346 {
347 XMMATRIX world = XMLoadFloat4x4(&e->World);
348 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
349
350 ObjectConstants objConstants;
351 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
352 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
353
354 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
355
356 // Next FrameResource need to be updated too.
357 e->NumFramesDirty--;
358 }
359 }
360}

362void Game::UpdateMaterialCBs(const GameTimer& gt)
363{
364 auto currMaterialCB = mCurrFrameResource->MaterialCB.get();
365 for (auto& e : mMaterials)
366 {
367 // Only update the cbuffer data if the constants have changed. If the cbuffer
368 // data changes, it needs to be updated for each FrameResource.
369 Material* mat = e.second.get();
370 if (mat->NumFramesDirty > 0)
371 {
372 XMMATRIX matTransform = XMLoadFloat4x4(&mat->MatTransform);
373
374 MaterialConstants matConstants;
375 matConstants.DiffuseAlbedo = mat->DiffuseAlbedo;
376 matConstants.FresnelR0 = mat->FresnelR0;
377 matConstants.Roughness = mat->Roughness;
378 XMStoreFloat4x4(&matConstants.MatTransform, XMMatrixTranspose(matTransform));
379
380 currMaterialCB->CopyData(mat->MatCBIndex, matConstants);
381
382 // Next FrameResource need to be updated too.
383 mat->NumFramesDirty--;
384 }

443void Game::BuildRootSignature()
444{
445 CD3DX12_DESCRIPTOR_RANGE texTable;
446 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
447
448 // Root parameter can be a table, root descriptor or root constants.
449 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
450
451 // Perfomance TIP: Order from most frequent to least frequent.
452 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
453 slotRootParameter[1].InitAsConstantBufferView(0);
454 slotRootParameter[2].InitAsConstantBufferView(1);
455 slotRootParameter[3].InitAsConstantBufferView(2);
456
457 auto staticSamplers = GetStaticSamplers();
458
459 // A root signature is an array of root parameters.
460 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
461 (UINT)staticSamplers.size(), staticSamplers.data(),
462 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
463
464 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
465 ComPtr<ID3DBlob> serializedRootSig = nullptr;
466 ComPtr<ID3DBlob> errorBlob = nullptr;
467 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
468 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
469
470 if (errorBlob != nullptr)
471 {
472 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
473 }
474 ThrowIfFailed(hr);
475
476 ThrowIfFailed(md3dDevice->CreateRootSignature(
477 0,
478 serializedRootSig->GetBufferPointer(),
479 serializedRootSig->GetBufferSize(),
480 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
481}

563void Game::BuildGroundGeometry()
564{
565 GeometryGenerator geoGen;
566 GeometryGenerator::MeshData ground = geoGen.CreateBox(20, 0.2, 20, 1);
567
568 std::vector<Vertex> vertices(ground.Vertices.size());
569 for (size_t i = 0; i < ground.Vertices.size(); ++i)
570 {
571 auto& p = ground.Vertices[i].Position;
572 vertices[i].Pos = p;
573 vertices[i].Normal = ground.Vertices[i].Normal;
574 vertices[i].TexC = ground.Vertices[i].TexC;
575 }
576
577 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
578
579 std::vector<std::uint16_t> indices = ground.GetIndices16();
580 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
581
582 auto geo = std::make_unique<MeshGeometry>();
583 geo->Name = "groundGeo";
584
585 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
586 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
587
588 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
589 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
590
591 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
592 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
593
594 geo->IndexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
595 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);
596
597 geo->VertexByteStride = sizeof(Vertex);
598 geo->VertexBufferByteSize = vbByteSize;
599 geo->IndexFormat = DXGI_FORMAT_R16_UINT;
600 geo->IndexBufferByteSize = ibByteSize;
601
602 SubmeshGeometry submesh;

703void Game::DrawRenderItems(ID3D12GraphicsCommandList* cmdList, const std::vector<RenderItem*>& ritems)
704{
705 UINT objCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(ObjectConstants));
706 UINT matCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(MaterialConstants));
707
708 auto objectCB = mCurrFrameResource->ObjectCB->Resource();
709 auto matCB = mCurrFrameResource->MaterialCB->Resource();
710
711 // For each render item...
712 for (size_t i = 0; i < ritems.size(); ++i)
713 {
714 auto ri = ritems[i];
715
716 cmdList->IASetVertexBuffers(0, 1, &ri->Geo->VertexBufferView());
717 cmdList->IASetIndexBuffer(&ri->Geo->IndexBufferView());
718 //
719 cmdList->IASetPrimitiveTopology(ri->PrimitiveType);
720
721 CD3DX12_GPU_DESCRIPTOR_HANDLE tex(mSrvDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
722 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);
723
724 D3D12_GPU_VIRTUAL_ADDRESS objCBAddress = objectCB->GetGPUVirtualAddress() + ri->ObjCBIndex * objCBByteSize;
725 D3D12_GPU_VIRTUAL_ADDRESS matCBAddress = matCB->GetGPUVirtualAddress() + ri->Mat->MatCBIndex * matCBByteSize;
726
727 cmdList->SetGraphicsRootDescriptorTable(0, tex);
728 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
729 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
730
731 cmdList->DrawIndexedInstanced(ri->IndexCount, 1, ri->StartIndexLocation, ri->BaseVertexLocation, 0);
732 }
