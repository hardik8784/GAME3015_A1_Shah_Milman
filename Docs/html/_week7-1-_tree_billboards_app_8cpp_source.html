25struct RenderItem
26{

187 return false;

189 // Reset the command list to prep for initialization commands.
190 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));

193 // so we have to query this information.

196 LoadTextures();
197 BuildRootSignature();
198 BuildDescriptorHeaps();
199 BuildShadersAndInputLayouts();
200 BuildTreeSpritesGeometry();
201 BuildMaterials();
202 BuildRenderItems();
203 BuildFrameResources();
204 BuildPSOs();
205
206 // Execute the initialization commands.
207 ThrowIfFailed(mCommandList->Close());
208 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
209 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
210
211 // Wait until initialization is complete.
212 FlushCommandQueue();

226void TreeBillboardsApp::Update(const GameTimer& gt)
227{
228 OnKeyboardInput(gt);
229 UpdateCamera(gt);
230
231 // Cycle through the circular frame resource array.
232 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
233 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
234
235 // Has the GPU finished processing the commands of the current frame resource?
236 // If not, wait until the GPU has completed commands up to this fence point.
237 if(mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
238 {
239 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
240 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
241 WaitForSingleObject(eventHandle, INFINITE);
242 CloseHandle(eventHandle);
243 }
244
245 UpdateObjectCBs(gt);
246 UpdateMaterialCBs(gt);
247 UpdateMainPassCB(gt);
248}

250void TreeBillboardsApp::Draw(const GameTimer& gt)
251{
252 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
253
254 // Reuse the memory associated with command recording.
255 // We can only reset when the associated command lists have finished execution on the GPU.
256 ThrowIfFailed(cmdListAlloc->Reset());
257
258 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
259 // Reusing the command list reuses memory.
260 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mPSOs["opaque"].Get()));
261
262 mCommandList->RSSetViewports(1, &mScreenViewport);
263 mCommandList->RSSetScissorRects(1, &mScissorRect);
264
265 // Indicate a state transition on the resource usage.
266 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
267 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
268
269 // Clear the back buffer and depth buffer.
270 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), (float*)&mMainPassCB.FogColor, 0, nullptr);
271 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
272
273 // Specify the buffers we are going to render to.
274 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
275
276 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
277 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
278
279 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
280
281 auto passCB = mCurrFrameResource->PassCB->Resource();
282 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
283
284
285 DrawRenderItems(mCommandList.Get(), mRitemLayer[(int)RenderLayer::AlphaTestedTreeSprites]);
286
287
288 // Indicate a state transition on the resource usage.
289 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
290 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
291
292 // Done recording commands.
293 ThrowIfFailed(mCommandList->Close());
294
295 // Add the command list to the queue for execution.
296 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
297 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
298
299 // Swap the back and front buffers
300 ThrowIfFailed(mSwapChain->Present(0, 0));
301 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
302
303 // Advance the fence value to mark commands up to this fence point.
304 mCurrFrameResource->Fence = ++mCurrentFence;
305
306 // Add an instruction to the command queue to set a new fence point.
307 // Because we are on the GPU timeline, the new fence point won't be
308 // set until the GPU finishes processing all the commands prior to this Signal().
309 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
310}

325void TreeBillboardsApp::OnMouseMove(WPARAM btnState, int x, int y)
326{
327 if((btnState & MK_LBUTTON) != 0)
328 {
329 // Make each pixel correspond to a quarter of a degree.
330 float dx = XMConvertToRadians(0.25f*static_cast<float>(x - mLastMousePos.x));
331 float dy = XMConvertToRadians(0.25f*static_cast<float>(y - mLastMousePos.y));
332
333 // Update angles based on input to orbit camera around box.
334 mTheta += dx;
335 mPhi += dy;
336
337 // Restrict the angle mPhi.
338 mPhi = MathHelper::Clamp(mPhi, 0.1f, MathHelper::Pi - 0.1f);
339 }
340 else if((btnState & MK_RBUTTON) != 0)
341 {
342 // Make each pixel correspond to 0.2 unit in the scene.
343 float dx = 0.2f*static_cast<float>(x - mLastMousePos.x);
344 float dy = 0.2f*static_cast<float>(y - mLastMousePos.y);
345
346 // Update the camera radius based on input.
347 mRadius += dx - dy;
348
349 // Restrict the radius.
350 mRadius = MathHelper::Clamp(mRadius, 5.0f, 150.0f);
351 }
352
353 mLastMousePos.x = x;
354 mLastMousePos.y = y;
355}

361void TreeBillboardsApp::UpdateCamera(const GameTimer& gt)
362{
363 // Convert Spherical to Cartesian coordinates.
364 mEyePos.x = mRadius*sinf(mPhi)*cosf(mTheta);
365 mEyePos.z = mRadius*sinf(mPhi)*sinf(mTheta);
366 mEyePos.y = mRadius*cosf(mPhi);
367
368 // Build the view matrix.
369 XMVECTOR pos = XMVectorSet(mEyePos.x, mEyePos.y, mEyePos.z, 1.0f);
370 XMVECTOR target = XMVectorZero();
371 XMVECTOR up = XMVectorSet(0.0f, 1.0f, 0.0f, 0.0f);
372
373 XMMATRIX view = XMMatrixLookAtLH(pos, target, up);
374 XMStoreFloat4x4(&mView, view);
375}

378void TreeBillboardsApp::UpdateObjectCBs(const GameTimer& gt)
379{
380 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
381 for(auto& e : mAllRitems)
382 {
383 // Only update the cbuffer data if the constants have changed.
384 // This needs to be tracked per frame resource.
385 if(e->NumFramesDirty > 0)
386 {
387 XMMATRIX world = XMLoadFloat4x4(&e->World);
388 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
389
390 ObjectConstants objConstants;
391 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
392 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
393
394 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
395
396 // Next FrameResource need to be updated too.
397 e->NumFramesDirty--;
398 }
399 }
400}

402void TreeBillboardsApp::UpdateMaterialCBs(const GameTimer& gt)

404 auto currMaterialCB = mCurrFrameResource->MaterialCB.get();
405 for(auto& e : mMaterials)
406 {
407 // Only update the cbuffer data if the constants have changed. If the cbuffer
408 // data changes, it needs to be updated for each FrameResource.
409 Material* mat = e.second.get();
410 if(mat->NumFramesDirty > 0)
411 {

413
414 MaterialConstants matConstants;
415 matConstants.DiffuseAlbedo = mat->DiffuseAlbedo;
416 matConstants.FresnelR0 = mat->FresnelR0;
417 matConstants.Roughness = mat->Roughness;
418 XMStoreFloat4x4(&matConstants.MatTransform, XMMatrixTranspose(matTransform));
419
420 currMaterialCB->CopyData(mat->MatCBIndex, matConstants);
421
422 // Next FrameResource need to be updated too.
423 mat->NumFramesDirty--;
424 }

428void TreeBillboardsApp::UpdateMainPassCB(const GameTimer& gt)
429{
430 XMMATRIX view = XMLoadFloat4x4(&mView);
431 XMMATRIX proj = XMLoadFloat4x4(&mProj);
432
433 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
434 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
435 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
436 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
437
438 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
439 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
440 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
441 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
442 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
443 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
444 mMainPassCB.EyePosW = mEyePos;
445 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
446 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
447 mMainPassCB.NearZ = 1.0f;
448 mMainPassCB.FarZ = 1000.0f;
449 mMainPassCB.TotalTime = gt.TotalTime();
450 mMainPassCB.DeltaTime = gt.DeltaTime();
451 mMainPassCB.AmbientLight = { 0.25f, 0.25f, 0.35f, 1.0f };
452 mMainPassCB.Lights[0].Direction = { 0.57735f, -0.57735f, 0.57735f };
453 mMainPassCB.Lights[0].Strength = { 0.6f, 0.6f, 0.6f };
454 mMainPassCB.Lights[1].Direction = { -0.57735f, -0.57735f, 0.57735f };
455 mMainPassCB.Lights[1].Strength = { 0.3f, 0.3f, 0.3f };
456 mMainPassCB.Lights[2].Direction = { 0.0f, -0.707f, -0.707f };
457 mMainPassCB.Lights[2].Strength = { 0.15f, 0.15f, 0.15f };
458
459 auto currPassCB = mCurrFrameResource->PassCB.get();
460 currPassCB->CopyData(0, mMainPassCB);
461}

479void TreeBillboardsApp::BuildRootSignature()
480{
481 CD3DX12_DESCRIPTOR_RANGE texTable;
482 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
483
484 // Root parameter can be a table, root descriptor or root constants.
485 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
486
487 // Perfomance TIP: Order from most frequent to least frequent.
488 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
489 slotRootParameter[1].InitAsConstantBufferView(0);
490 slotRootParameter[2].InitAsConstantBufferView(1);
491 slotRootParameter[3].InitAsConstantBufferView(2);
492
493 auto staticSamplers = GetStaticSamplers();
494
495 // A root signature is an array of root parameters.
496 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
497 (UINT)staticSamplers.size(), staticSamplers.data(),
498 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
499
500 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
501 ComPtr<ID3DBlob> serializedRootSig = nullptr;
502 ComPtr<ID3DBlob> errorBlob = nullptr;
503 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
504 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
505
506 if(errorBlob != nullptr)
507 {
508 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
509 }
510 ThrowIfFailed(hr);
511
512 ThrowIfFailed(md3dDevice->CreateRootSignature(
513 0,
514 serializedRootSig->GetBufferPointer(),
515 serializedRootSig->GetBufferSize(),
516 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
517}

581void TreeBillboardsApp::BuildTreeSpritesGeometry()
582{
583 struct TreeSpriteVertex
584 {
585 XMFLOAT3 Pos;
586 XMFLOAT2 Size;
587 };
588
589 static const int treeCount = 4;
590 std::array<TreeSpriteVertex, 4> vertices;
591 for(UINT i = 0; i < treeCount; ++i)
592 {
593 float x = MathHelper::RandF(-45.0f, 45.0f);
594 float z = MathHelper::RandF(-45.0f, 45.0f);
595 float y = GetHillsHeight(x, z);
596
597 // Move tree slightly above land height.
598 y += 8.0f;
599
600 vertices[i].Pos = XMFLOAT3(x, y, z);
601 vertices[i].Size = XMFLOAT2(20.0f, 20.0f);
602 }
603
604 std::array<std::uint16_t, 4> indices =
605 {
606 0, 1, 2, 3,
607 //4, 5, 6, 7,
608 //8, 9, 10, 11, 12, 13, 14, 15
609 };
610
611 const UINT vbByteSize = (UINT)vertices.size() * sizeof(TreeSpriteVertex);
612 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
613
614 auto geo = std::make_unique<MeshGeometry>();
615 geo->Name = "treeSpritesGeo";
616
617 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
618 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
619
620 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
621 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
622
623 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
624 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);

732void TreeBillboardsApp::DrawRenderItems(ID3D12GraphicsCommandList* cmdList, const std::vector<RenderItem*>& ritems)
733{
734 UINT objCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(ObjectConstants));
735 UINT matCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(MaterialConstants));
736
737 auto objectCB = mCurrFrameResource->ObjectCB->Resource();
738 auto matCB = mCurrFrameResource->MaterialCB->Resource();
739
740 // For each render item...
741 for(size_t i = 0; i < ritems.size(); ++i)
742 {
743 auto ri = ritems[i];
744
745 cmdList->IASetVertexBuffers(0, 1, &ri->Geo->VertexBufferView());
746 cmdList->IASetIndexBuffer(&ri->Geo->IndexBufferView());
747 cmdList->IASetPrimitiveTopology(ri->PrimitiveType);
748
749 CD3DX12_GPU_DESCRIPTOR_HANDLE tex(mSrvDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
750 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);
751
752 D3D12_GPU_VIRTUAL_ADDRESS objCBAddress = objectCB->GetGPUVirtualAddress() + ri->ObjCBIndex*objCBByteSize;
753 D3D12_GPU_VIRTUAL_ADDRESS matCBAddress = matCB->GetGPUVirtualAddress() + ri->Mat->MatCBIndex*matCBByteSize;
754
755 cmdList->SetGraphicsRootDescriptorTable(0, tex);
756 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
757 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
758
759 cmdList->DrawIndexedInstanced(ri->IndexCount, 1, ri->StartIndexLocation, ri->BaseVertexLocation, 0);
760 }
